# P3.1 — AI writing assistant

## Priority: P3 — strategic differentiator

## Problem

Creating HMS documents from scratch is time-consuming. Users stare at a blank
template not knowing how to phrase the required content. Notion AI and Confluence
AI both offer in-editor assistance. For a compliance platform the opportunity is
higher: the AI can be pre-seeded with Norwegian regulatory context.

## Acceptance criteria

- [ ] A "Skriv med AI" button appears in the toolbar of text and heading blocks
- [ ] Clicking the button opens a small prompt input inside the block card
- [ ] The user types a brief instruction (e.g. "Skriv en rutine for håndtering av avvik")
- [ ] The AI response streams into the block's text content
- [ ] The user can accept (keep text) or discard (revert to previous content)
- [ ] The AI system prompt includes: the page's legal references, the document
  template type, and a Norwegian language instruction
- [ ] The call is made via a Supabase Edge Function (API key never exposed to client)
- [ ] `npm run build` exits 0, zero TS errors

## Stack

- Claude API via `@anthropic-ai/sdk` (latest version)
- Supabase Edge Function as the secure proxy
- Streaming response via Server-Sent Events

## Edge Function

```ts
// supabase/functions/wiki-ai-assist/index.ts
import Anthropic from 'npm:@anthropic-ai/sdk'

const anthropic = new Anthropic({ apiKey: Deno.env.get('ANTHROPIC_API_KEY')! })

Deno.serve(async (req) => {
  if (req.method !== 'POST') return new Response('Method Not Allowed', { status: 405 })

  const { instruction, legalRefs, templateType } = await req.json()

  const systemPrompt = [
    'Du er en ekspert på norsk HMS-lovgivning og skriver klart og presist på norsk bokmål.',
    'Dokumenttype: ' + (templateType ?? 'generell HMS-rutine'),
    legalRefs?.length ? 'Relevante lovhenvisninger: ' + legalRefs.join(', ') : '',
    'Svar kun med dokumentinnhold — ingen innledning eller forklaring.',
  ].filter(Boolean).join('\n')

  const stream = anthropic.messages.stream({
    model: 'claude-sonnet-4-6',
    max_tokens: 1024,
    system: systemPrompt,
    messages: [{ role: 'user', content: instruction }],
  })

  const encoder = new TextEncoder()
  const readable = new ReadableStream({
    async start(controller) {
      for await (const chunk of stream) {
        if (
          chunk.type === 'content_block_delta' &&
          chunk.delta.type === 'text_delta'
        ) {
          controller.enqueue(encoder.encode(chunk.delta.text))
        }
      }
      controller.close()
    },
  })

  return new Response(readable, {
    headers: {
      'Content-Type': 'text/plain; charset=utf-8',
      'Transfer-Encoding': 'chunked',
    },
  })
})
```

Set `ANTHROPIC_API_KEY` as a Supabase Edge Function secret (never commit the key).

## Client hook

```ts
// src/hooks/useWikiAiAssist.ts
import { useCallback, useState } from 'react'
import { supabase } from '../lib/supabaseClient'  // verify import path

export function useWikiAiAssist() {
  const [streaming, setStreaming] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const assist = useCallback(async (
    instruction: string,
    legalRefs: string[],
    templateType: string,
    onChunk: (text: string) => void,
  ) => {
    setStreaming(true)
    setError(null)
    try {
      const { data: { session } } = await supabase.auth.getSession()
      const res = await fetch(
        `${import.meta.env.VITE_SUPABASE_URL}/functions/v1/wiki-ai-assist`,
        {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${session?.access_token ?? ''}`,
          },
          body: JSON.stringify({ instruction, legalRefs, templateType }),
        }
      )
      if (!res.ok) throw new Error(`Feil fra AI-tjenesten (${res.status})`)
      const reader = res.body?.getReader()
      const decoder = new TextDecoder()
      if (!reader) throw new Error('Ingen respons fra tjenesten')
      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        onChunk(decoder.decode(value))
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Ukjent feil')
    } finally {
      setStreaming(false)
    }
  }, [])

  return { assist, streaming, error }
}
```
