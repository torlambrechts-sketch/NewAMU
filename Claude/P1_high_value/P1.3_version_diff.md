# P1.3 — Version diff / compare view

## Priority: P1

## Problem

Version history shows timestamps but no way to see what changed. For
compliance-heavy HMS documents this is critical for auditors and inspectors.

## Acceptance criteria

- [ ] Each version row in `WikiPageView` has a "Sammenlign" button
- [ ] Clicking compares that version against the current published version
- [ ] Diff: deleted text red strikethrough, added text green
- [ ] "Lukk" dismisses the diff panel
- [ ] No TypeScript errors

## WikiVersionDiff component

```tsx
import DiffMatchPatch from 'diff-match-patch'
import type { ContentBlock, WikiPageVersionSnapshot } from '../../types/documents'

function blockToText(block: ContentBlock): string {
  switch (block.kind) {
    case 'text': return block.body.replace(/<[^>]+>/g, '')
    case 'heading': return block.text
    case 'alert': return block.text
    case 'law_ref': return `${block.ref}: ${block.description}`
    case 'image': return block.caption ?? ''
    case 'divider': return '───'
    case 'module': return `[${block.moduleName}]`
    default: return ''
  }
}

export function WikiVersionDiff({ versionA, versionB }: {
  versionA: WikiPageVersionSnapshot; versionB: WikiPageVersionSnapshot
}) {
  const dmp = new DiffMatchPatch()
  const diffs = dmp.diff_main(
    versionA.blocks.map(blockToText).join('\n\n'),
    versionB.blocks.map(blockToText).join('\n\n')
  )
  dmp.diff_cleanupSemantic(diffs)
  return (
    <div className="rounded-xl border border-neutral-200 bg-neutral-50 p-4 font-mono text-sm">
      <p className="mb-3 font-sans text-xs font-semibold uppercase text-neutral-500">
        v{versionA.version} → v{versionB.version}
      </p>
      <div className="whitespace-pre-wrap">
        {diffs.map(([op, text], i) => (
          op === 0 ? <span key={i} className="text-neutral-700">{text}</span>
          : op === -1 ? <span key={i} className="bg-red-100 text-red-700 line-through">{text}</span>
          : <span key={i} className="bg-emerald-100 text-emerald-800">{text}</span>
        ))}
      </div>
    </div>
  )
}
```

## Wiring in WikiPageView

```tsx
const [diffVersion, setDiffVersion] = useState<WikiPageVersionSnapshot | null>(null)

// In version list:
<button onClick={() => setDiffVersion(v)} className="text-xs text-[#1a3d32] underline">
  Sammenlign
</button>

// Below content:
{diffVersion && currentVersion && (
  <div className="mt-8">
    <div className="mb-2 flex items-center justify-between">
      <h3 className="text-sm font-semibold">Endringer</h3>
      <button className="text-xs text-neutral-500" onClick={() => setDiffVersion(null)}>Lukk</button>
    </div>
    <WikiVersionDiff versionA={diffVersion} versionB={currentVersion} />
  </div>
)}
```
