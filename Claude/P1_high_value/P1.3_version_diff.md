# P1.3 — Version diff / compare view

## Priority: P1

## Problem

Version history in `WikiPageView` shows timestamps and version numbers but no
way to see what actually changed between two versions. For compliance-heavy HMS
documents, inspectors and document owners need to see exactly what changed.

## Acceptance criteria

- [ ] In the version history sidebar, each version row has a "Sammenlign" button
- [ ] Clicking "Sammenlign" on version N compares it against the current published version
- [ ] The diff view renders deleted text in red strikethrough, added text in green
- [ ] Diff is shown at the block level: which blocks were added, removed, or changed
- [ ] The diff view is rendered in a modal or a full-width panel below the content
- [ ] A "Lukk" button dismisses the diff view
- [ ] No TypeScript errors

## Implementation

### Block-level diff logic

Convert each `ContentBlock` to a plain text string for comparison:

```tsx
import DiffMatchPatch from 'diff-match-patch'

function blockToText(block: ContentBlock): string {
  switch (block.kind) {
    case 'text': return block.body.replace(/<[^>]+>/g, '') // strip HTML tags
    case 'heading': return block.text
    case 'alert': return block.text
    case 'law_ref': return `${block.ref} ${block.description}`
    case 'image': return block.caption ?? block.alt ?? ''
    case 'divider': return '---'
    case 'module': return `[modul: ${block.moduleName}]`
    default: return ''
  }
}

function computeDiff(oldText: string, newText: string) {
  const dmp = new DiffMatchPatch()
  return dmp.diff_main(oldText, newText)
}
```

### WikiVersionDiff component

```tsx
// src/components/documents/WikiVersionDiff.tsx
import DiffMatchPatch from 'diff-match-patch'
import type { ContentBlock, WikiPageVersionSnapshot } from '../../types/documents'

type Diff = [number, string]  // [-1 delete, 0 equal, 1 insert]

function blockToText(block: ContentBlock): string {
  switch (block.kind) {
    case 'text': return block.body.replace(/<[^>]+>/g, '')
    case 'heading': return block.text
    case 'alert': return block.text
    case 'law_ref': return `${block.ref}: ${block.description}`
    case 'image': return block.caption ?? ''
    case 'divider': return '───'
    case 'module': return `[${block.moduleName}]`
    default: return ''
  }
}

export function WikiVersionDiff({
  versionA,
  versionB,
}: {
  versionA: WikiPageVersionSnapshot
  versionB: WikiPageVersionSnapshot
}) {
  const dmp = new DiffMatchPatch()
  const oldText = versionA.blocks.map(blockToText).join('\n\n')
  const newText = versionB.blocks.map(blockToText).join('\n\n')
  const diffs: Diff[] = dmp.diff_main(oldText, newText)
  dmp.diff_cleanupSemantic(diffs)

  return (
    <div className="rounded-xl border border-neutral-200 bg-neutral-50 p-4 font-mono text-sm leading-relaxed">
      <p className="mb-3 font-sans text-xs font-semibold uppercase tracking-wide text-neutral-500">
        v{versionA.version} → v{versionB.version}
      </p>
      <div className="whitespace-pre-wrap break-words">
        {diffs.map(([op, text], i) => {
          if (op === 0) return <span key={i} className="text-neutral-700">{text}</span>
          if (op === -1) return <span key={i} className="bg-red-100 text-red-700 line-through">{text}</span>
          return <span key={i} className="bg-emerald-100 text-emerald-800">{text}</span>
        })}
      </div>
    </div>
  )
}
```

### Add "Sammenlign" button to version history

In `WikiPageView.tsx`, find the version history list. For each past version row
add:

```tsx
<button
  className="text-xs text-[#1a3d32] underline hover:no-underline"
  onClick={() => setDiffVersion(v)}  // v is the WikiPageVersionSnapshot
>
  Sammenlign
</button>
```

Add state: `const [diffVersion, setDiffVersion] = useState<WikiPageVersionSnapshot | null>(null)`

Render the diff panel below the page content when `diffVersion` is set:

```tsx
{diffVersion && currentVersion && (
  <div className="mt-8">
    <div className="mb-2 flex items-center justify-between">
      <h3 className="text-sm font-semibold text-neutral-800">Endringer</h3>
      <button
        className="text-xs text-neutral-500 hover:text-neutral-700"
        onClick={() => setDiffVersion(null)}
      >
        Lukk
      </button>
    </div>
    <WikiVersionDiff versionA={diffVersion} versionB={currentVersion} />
  </div>
)}
```

`currentVersion` is the latest `WikiPageVersionSnapshot` from `versionsForPage`.
