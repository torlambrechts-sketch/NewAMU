# P1.1 — Full-text search across all documents

## Priority: P1 — highest impact addition

## Problem

There is no way to search across wiki pages. Users must navigate folder by
folder. Every major wiki tool leads with search.

## Acceptance criteria

- [ ] `Cmd+K` / `Ctrl+K` opens a command-palette search modal from anywhere in the documents module
- [ ] Search matches page `title`, `summary`, and plain text content of `blocks`
- [ ] Results show: page title, space name, status badge, last updated date
- [ ] Clicking a result navigates to `/documents/page/:pageId`
- [ ] Results appear within 300 ms (debounced)
- [ ] Empty state: "Ingen resultater for \"<query>\""
- [ ] Only pages the current user has access to are returned (RLS enforced)
- [ ] `npm run build` exits 0, zero TS errors

## Database migration

```sql
ALTER TABLE wiki_pages
  ADD COLUMN IF NOT EXISTS search_vector tsvector
  GENERATED ALWAYS AS (
    to_tsvector('norwegian', coalesce(title, '') || ' ' || coalesce(summary, ''))
  ) STORED;

CREATE INDEX IF NOT EXISTS wiki_pages_search_vector_idx
  ON wiki_pages USING GIN (search_vector);

CREATE OR REPLACE FUNCTION search_wiki_pages(
  p_org_id uuid, p_query text, p_limit int DEFAULT 20
)
RETURNS TABLE (id uuid, title text, summary text, status text,
               space_id uuid, updated_at timestamptz, rank float4)
LANGUAGE sql STABLE SECURITY INVOKER AS $$
  SELECT p.id, p.title, p.summary, p.status, p.space_id, p.updated_at,
    ts_rank(p.search_vector, to_tsquery('norwegian', p_query || ':*')) AS rank
  FROM wiki_pages p
  WHERE p.org_id = p_org_id
    AND p.search_vector @@ to_tsquery('norwegian', p_query || ':*')
  ORDER BY rank DESC LIMIT p_limit;
$$;
```

`SECURITY INVOKER` ensures RLS applies to the calling user.

## useDocumentSearch hook

```tsx
import { useCallback, useEffect, useState } from 'react'
import { supabase } from '../lib/supabaseClient'  // verify import path

export type SearchResult = {
  id: string; title: string; summary: string | null; status: string
  spaceId: string; updatedAt: string; rank: number
}

export function useDocumentSearch(orgId: string) {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<SearchResult[]>([])
  const [loading, setLoading] = useState(false)

  const search = useCallback(async (q: string) => {
    if (!q.trim() || q.trim().length < 2) { setResults([]); return }
    setLoading(true)
    try {
      const { data, error } = await supabase.rpc('search_wiki_pages',
        { p_org_id: orgId, p_query: q.trim(), p_limit: 20 })
      if (error) throw error
      setResults((data ?? []).map((r: Record<string, unknown>) => ({
        id: r.id as string, title: r.title as string,
        summary: r.summary as string | null, status: r.status as string,
        spaceId: r.space_id as string, updatedAt: r.updated_at as string,
        rank: r.rank as number,
      })))
    } finally { setLoading(false) }
  }, [orgId])

  // Debounce: 250 ms
  useEffect(() => {
    const id = setTimeout(() => search(query), 250)
    return () => clearTimeout(id)
  }, [query, search])

  return { query, setQuery, results, loading }
}
```

## Cmd+K wiring (DocumentsModuleShellLayout)

```tsx
const [searchOpen, setSearchOpen] = useState(false)

useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault(); setSearchOpen(true)
    }
  }
  window.addEventListener('keydown', handler)
  return () => window.removeEventListener('keydown', handler)
}, [])
```
