# P1.1 — Full-text search across all documents

## Priority: P1 — highest impact addition

## Problem

There is no way to search across wiki pages. Users must navigate folder by folder
to find a document. For a platform with 50+ HMS documents this is a critical
usability gap. Every major wiki tool (Confluence, Notion, Guru) leads with search.

## Acceptance criteria

- [ ] A search input is accessible from the documents hub (`/documents`)
- [ ] `Cmd+K` / `Ctrl+K` opens a command-palette style search modal from anywhere in the documents module
- [ ] Search matches page `title`, `summary`, and plain text content of `blocks`
- [ ] Results show: page title, space name, status badge, last updated date
- [ ] Clicking a result navigates to `/documents/page/:pageId`
- [ ] Results appear within 300 ms of the user stopping typing (debounced)
- [ ] Empty state: "Ingen resultater for \"<query>\""
- [ ] Only pages the current user has access to are returned (RLS enforced)
- [ ] `npm run build` exits 0, zero TS errors

## Database migration

File name: `YYYYMMDDHHMMSS_wiki_search_tsvector.sql`
(Replace YYYYMMDDHHMMSS with the current timestamp at time of writing.)

```sql
-- Add a generated tsvector column for full-text search
ALTER TABLE wiki_pages
  ADD COLUMN IF NOT EXISTS search_vector tsvector
  GENERATED ALWAYS AS (
    to_tsvector('norwegian',
      coalesce(title, '') || ' ' || coalesce(summary, '')
    )
  ) STORED;

-- GIN index for fast search
CREATE INDEX IF NOT EXISTS wiki_pages_search_vector_idx
  ON wiki_pages USING GIN (search_vector);

-- RPC function that respects RLS (called from the client)
CREATE OR REPLACE FUNCTION search_wiki_pages(
  p_org_id uuid,
  p_query text,
  p_limit int DEFAULT 20
)
RETURNS TABLE (
  id uuid,
  title text,
  summary text,
  status text,
  space_id uuid,
  updated_at timestamptz,
  rank float4
)
LANGUAGE sql
STABLE
SECURITY INVOKER  -- runs as the calling user so RLS applies
AS $$
  SELECT
    p.id,
    p.title,
    p.summary,
    p.status,
    p.space_id,
    p.updated_at,
    ts_rank(p.search_vector, to_tsquery('norwegian', p_query || ':*')) AS rank
  FROM wiki_pages p
  WHERE
    p.org_id = p_org_id
    AND p.search_vector @@ to_tsquery('norwegian', p_query || ':*')
  ORDER BY rank DESC
  LIMIT p_limit;
$$;
```

Note: `blocks` JSONB is intentionally excluded from the DB tsvector to keep
the index small. Block text can be searched client-side as a secondary pass if
needed.

## useDocumentSearch hook

```tsx
// src/hooks/useDocumentSearch.ts
import { useCallback, useEffect, useState } from 'react'
import { useDebounce } from 'use-debounce'  // already in package.json — verify first
import { supabase } from '../lib/supabaseClient'  // verify actual import path

export type SearchResult = {
  id: string
  title: string
  summary: string | null
  status: string
  spaceId: string
  updatedAt: string
  rank: number
}

export function useDocumentSearch(orgId: string) {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<SearchResult[]>([])
  const [loading, setLoading] = useState(false)
  const [debouncedQuery] = useDebounce(query, 250)

  const search = useCallback(async (q: string) => {
    if (!q.trim() || q.trim().length < 2) {
      setResults([])
      return
    }
    setLoading(true)
    try {
      const { data, error } = await supabase.rpc('search_wiki_pages', {
        p_org_id: orgId,
        p_query: q.trim(),
        p_limit: 20,
      })
      if (error) throw error
      setResults(
        (data ?? []).map((r: Record<string, unknown>) => ({
          id: r.id as string,
          title: r.title as string,
          summary: r.summary as string | null,
          status: r.status as string,
          spaceId: r.space_id as string,
          updatedAt: r.updated_at as string,
          rank: r.rank as number,
        }))
      )
    } finally {
      setLoading(false)
    }
  }, [orgId])

  useEffect(() => { search(debouncedQuery) }, [debouncedQuery, search])

  return { query, setQuery, results, loading }
}
```

Verify whether `use-debounce` is already in `package.json`. If not, use a
manual `useEffect` with `setTimeout`/`clearTimeout` instead.

## Search modal UI

```tsx
// src/components/documents/DocumentSearchModal.tsx
import { useEffect, useRef } from 'react'
import { Link } from 'react-router-dom'
import { Search, X } from 'lucide-react'
import { useDocumentSearch } from '../../hooks/useDocumentSearch'
import { useDocuments } from '../../hooks/useDocuments'

export function DocumentSearchModal({
  open,
  onClose,
  orgId,
}: {
  open: boolean
  onClose: () => void
  orgId: string
}) {
  const { query, setQuery, results, loading } = useDocumentSearch(orgId)
  const docs = useDocuments()
  const inputRef = useRef<HTMLInputElement>(null)

  useEffect(() => {
    if (open) setTimeout(() => inputRef.current?.focus(), 50)
  }, [open])

  useEffect(() => {
    if (!open) return
    const handler = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose() }
    window.addEventListener('keydown', handler)
    return () => window.removeEventListener('keydown', handler)
  }, [open, onClose])

  if (!open) return null

  return (
    <div className="fixed inset-0 z-[70] flex items-start justify-center bg-black/50 pt-20">
      <div className="w-full max-w-xl overflow-hidden rounded-2xl bg-white shadow-2xl">
        <div className="flex items-center gap-3 border-b border-neutral-200 px-4 py-3">
          <Search className="size-5 shrink-0 text-neutral-400" />
          <input
            ref={inputRef}
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            placeholder="Søk i dokumenter…"
            className="flex-1 bg-transparent text-sm outline-none placeholder:text-neutral-400"
          />
          <button onClick={onClose} aria-label="Lukk søk">
            <X className="size-5 text-neutral-400 hover:text-neutral-700" />
          </button>
        </div>
        <div className="max-h-96 overflow-y-auto">
          {loading && (
            <p className="px-4 py-8 text-center text-sm text-neutral-500">Søker…</p>
          )}
          {!loading && query.length >= 2 && results.length === 0 && (
            <p className="px-4 py-8 text-center text-sm text-neutral-500">
              Ingen resultater for &laquo;{query}&raquo;
            </p>
          )}
          {!loading && results.map((r) => {
            const space = docs.spaces.find((s) => s.id === r.spaceId)
            return (
              <Link
                key={r.id}
                to={`/documents/page/${r.id}`}
                onClick={onClose}
                className="flex flex-col gap-0.5 px-4 py-3 hover:bg-neutral-50"
              >
                <span className="text-sm font-medium text-neutral-900">{r.title}</span>
                <span className="text-xs text-neutral-500">
                  {space?.title ?? 'Ukjent mappe'} ·{' '}
                  {new Date(r.updatedAt).toLocaleDateString('no-NO')}
                </span>
              </Link>
            )
          })}
        </div>
      </div>
    </div>
  )
}
```

## Cmd+K wiring

In `modules/documents/DocumentsModuleShellLayout.tsx` (read the file first):

```tsx
const [searchOpen, setSearchOpen] = useState(false)

useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault()
      setSearchOpen(true)
    }
  }
  window.addEventListener('keydown', handler)
  return () => window.removeEventListener('keydown', handler)
}, [])
```

Mount `<DocumentSearchModal open={searchOpen} onClose={() => setSearchOpen(false)} orgId={orgId} />`
near the root of the layout JSX.
