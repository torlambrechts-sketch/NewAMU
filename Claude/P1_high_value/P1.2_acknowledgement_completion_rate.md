## Helper: audience count

```tsx
function getAudienceCount(page: WikiPage, members: Member[]): number {
  switch (page.acknowledgementAudience ?? 'all_employees') {
    case 'all_employees': return members.length
    case 'leaders_only': return members.filter((m) => m.role === 'admin' || m.role === 'leader').length
    case 'safety_reps_only': return members.filter((m) => m.role === 'safety_rep').length
    case 'department': return members.filter((m) => m.departmentId === page.acknowledgementDepartmentId).length
    default: return members.length
  }
}
```

## Progress bar component

```tsx
function AckProgressBar({ signed, total }: { signed: number; total: number }) {
  const pct = total === 0 ? 0 : Math.round((signed / total) * 100)
  const colour = pct >= 90 ? 'bg-emerald-500' : pct >= 60 ? 'bg-amber-400' : 'bg-red-500'
  return (
    <div className="flex items-center gap-2">
      <div className="h-2 w-24 overflow-hidden rounded-full bg-neutral-200">
        <div className={`h-full rounded-full ${colour}`} style={{ width: `${pct}%` }} />
      </div>
      <span className={`text-xs font-medium ${
        pct >= 90 ? 'text-emerald-700' : pct >= 60 ? 'text-amber-700' : 'text-red-700'
      }`}>{signed} av {total} ({pct} %)</span>
    </div>
  )
}
```
