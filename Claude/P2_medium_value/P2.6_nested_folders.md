# P2.6 — Nested folder hierarchy

## Priority: P2

## Problem

All wiki spaces (folders) are flat — one level only. Organisations with many
departments and procedure categories need sub-folders. Confluence and Notion
both support multi-level nesting.

- [ ] The sidebar navigation renders spaces as an expandable tree
- [ ] Creating a new space allows selecting a parent space
- [ ] Moving a space to a different parent is supported via the folder settings
- [ ] Maximum nesting depth is 3 levels (enforced client-side and DB check)
- [ ] Breadcrumb navigation on `WikiSpaceView` reflects the full path
- [ ] `npm run build` exits 0, zero TS errors

## Database migration

```sql
ALTER TABLE wiki_spaces
  ADD COLUMN IF NOT EXISTS parent_space_id uuid
    REFERENCES wiki_spaces(id) ON DELETE SET NULL;

-- Prevent cycles at depth > 3 (check constraint via function)
CREATE OR REPLACE FUNCTION wiki_space_depth(p_id uuid)
RETURNS int
LANGUAGE sql STABLE AS $$
  WITH RECURSIVE tree AS (
    SELECT id, parent_space_id, 1 AS depth
    FROM wiki_spaces WHERE id = p_id
    UNION ALL
    SELECT s.id, s.parent_space_id, t.depth + 1
    FROM wiki_spaces s JOIN tree t ON s.id = t.parent_space_id
  )
  SELECT max(depth) FROM tree;
$$;

ALTER TABLE wiki_spaces
  ADD CONSTRAINT wiki_spaces_max_depth
    CHECK (wiki_space_depth(id) <= 3);
```

## Tree navigation helper

Add a pure helper function (not exported) in `documentsNav.ts` or a new
`src/lib/wikiSpaceTree.ts`:

```ts
export type SpaceTreeNode = WikiSpace & { children: SpaceTreeNode[] }

export function buildSpaceTree(spaces: WikiSpace[]): SpaceTreeNode[] {
  const map = new Map<string, SpaceTreeNode>()
  spaces.forEach((s) => map.set(s.id, { ...s, children: [] }))
  const roots: SpaceTreeNode[] = []
  map.forEach((node) => {
    if (node.parentSpaceId && map.has(node.parentSpaceId)) {
      map.get(node.parentSpaceId)!.children.push(node)
    } else {
      roots.push(node)
    }
  })
  return roots
}
```

## Breadcrumb helper

```ts
export function getSpacePath(spaceId: string, spaces: WikiSpace[]): WikiSpace[] {
  const path: WikiSpace[] = []
  let current = spaces.find((s) => s.id === spaceId)
  while (current) {
    path.unshift(current)
    current = current.parentSpaceId
      ? spaces.find((s) => s.id === current!.parentSpaceId)
      : undefined
  }
  return path
}
```

Render in `WikiSpaceView.tsx`:

```tsx
const path = getSpacePath(space.id, docs.spaces)
<nav aria-label="Brødsmule">
  {path.map((s, i) => (
    <span key={s.id}>
      {i > 0 && <span className="mx-1 text-neutral-400">/</span>}
      {i < path.length - 1 ? (
        <Link to={`/documents/space/${s.id}`} className="text-[#1a3d32] hover:underline">{s.title}</Link>
      ) : (
        <span className="font-semibold text-neutral-800">{s.title}</span>
      )}
    </span>
  ))}
</nav>
```
