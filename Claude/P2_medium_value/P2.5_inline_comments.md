# P2.5 — Inline comments / review annotations

## Priority: P2

## Problem

Safety representatives and document reviewers need to annotate published pages.
There is no mechanism for a reader to highlight a paragraph and say "dette
stemmer ikke med gjeldende praksis" without editing the document itself.
Confluence, Guru, and Document360 all support inline comments as a core feature.

## Acceptance criteria

- [ ] Readers can click a comment icon next to any block to open a comment thread
- [ ] Posting a comment saves it to a new `wiki_page_comments` table
- [ ] Comments are visible to all users with `documents.view` permission for the page
- [ ] Comments can be marked as resolved (hidden from default view)
- [ ] Resolved comments can be toggled visible via "Vis løste kommentarer"
- [ ] The author and timestamp are shown on each comment
- [ ] Comment authors can delete their own comments
- [ ] `npm run build` exits 0, zero TS errors

## Database migration

```sql
CREATE TABLE IF NOT EXISTS wiki_page_comments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id uuid NOT NULL REFERENCES organisations(id) ON DELETE CASCADE,
  page_id uuid NOT NULL REFERENCES wiki_pages(id) ON DELETE CASCADE,
  block_index int NOT NULL,
  body text NOT NULL CHECK (char_length(body) BETWEEN 1 AND 2000),
  author_id uuid NOT NULL REFERENCES auth.users(id),
  author_name text NOT NULL,
  resolved boolean NOT NULL DEFAULT false,
  created_at timestamptz NOT NULL DEFAULT now()
);

ALTER TABLE wiki_page_comments ENABLE ROW LEVEL SECURITY;

-- Org members can read comments on pages they can access
CREATE POLICY "org_member_read" ON wiki_page_comments
  FOR SELECT USING (
    org_id IN (
      SELECT org_id FROM org_members WHERE user_id = auth.uid()
    )
  );

-- Any org member can insert
CREATE POLICY "org_member_insert" ON wiki_page_comments
  FOR INSERT WITH CHECK (
    author_id = auth.uid() AND
    org_id IN (SELECT org_id FROM org_members WHERE user_id = auth.uid())
  );

-- Author can update (resolve) or delete their own comment
CREATE POLICY "author_update" ON wiki_page_comments
  FOR UPDATE USING (author_id = auth.uid());

CREATE POLICY "author_delete" ON wiki_page_comments
  FOR DELETE USING (author_id = auth.uid());
```

## useWikiComments hook

```ts
// src/hooks/useWikiComments.ts
import { useCallback, useEffect, useState } from 'react'
import { supabase } from '../lib/supabaseClient'  // verify import path
import type { WikiPageComment } from '../types/documents'

export function useWikiComments(pageId: string, orgId: string) {
  const [comments, setComments] = useState<WikiPageComment[]>([])

  const load = useCallback(async () => {
    const { data } = await supabase
      .from('wiki_page_comments')
      .select('*')
      .eq('page_id', pageId)
      .order('created_at', { ascending: true })
    setComments((data ?? []).map(mapRow))
  }, [pageId])

  useEffect(() => { load() }, [load])

  async function addComment(blockIndex: number, body: string, authorName: string) {
    await supabase.from('wiki_page_comments').insert({
      org_id: orgId, page_id: pageId, block_index: blockIndex,
      body, author_id: (await supabase.auth.getUser()).data.user?.id, author_name: authorName,
    })
    await load()
  }

  async function resolveComment(id: string) {
    await supabase.from('wiki_page_comments').update({ resolved: true }).eq('id', id)
    await load()
  }

  async function deleteComment(id: string) {
    await supabase.from('wiki_page_comments').delete().eq('id', id)
    await load()
  }

  return { comments, addComment, resolveComment, deleteComment }
}

function mapRow(r: Record<string, unknown>): WikiPageComment {
  return {
    id: r.id as string,
    pageId: r.page_id as string,
    blockIndex: r.block_index as number,
    body: r.body as string,
    authorId: r.author_id as string,
    authorName: r.author_name as string,
    resolved: r.resolved as boolean,
    createdAt: r.created_at as string,
  }
}
```
